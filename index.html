function doPost(e) { try { var data = JSON.parse(e.postData.contents); var action = data.action || 'upsert_trade'; var ss = SpreadsheetApp.getActiveSpreadsheet(); if (action === 'upsert_trade') { var sheet = ss.getSheetByName('🔧 Trade Register'); if (!sheet) throw new Error('Trade Register sheet not found'); var rows = sheet.getDataRange().getValues(); var matchRow = -1; for (var r = 1; r < rows.length; r++) { var tradeName = String(rows[r][1]).toLowerCase(); var incomingTrade = String(data.trade || '').toLowerCase(); if (tradeName.length > 0 && incomingTrade.length > 0) { var tradeShort = tradeName.substring(0, 6); var incomingShort = incomingTrade.substring(0, 6); if (tradeName.indexOf(incomingShort) > -1 || incomingTrade.indexOf(tradeShort) > -1) { matchRow = r + 1; break; } } } if (matchRow > 0) { if (data.company) sheet.getRange(matchRow, 3).setValue(data.company); if (data.contact) sheet.getRange(matchRow, 4).setValue(data.contact); if (data.phone) sheet.getRange(matchRow, 5).setValue(data.phone); if (data.email) sheet.getRange(matchRow, 6).setValue(data.email); if (data.status) sheet.getRange(matchRow, 7).setValue(data.status); if (data.date) sheet.getRange(matchRow, 8).setValue(data.date); sheet.getRange(matchRow, 9).setValue('Yes'); if (data.amount_inc_gst) sheet.getRange(matchRow, 10).setValue(data.amount_inc_gst); sheet.getRange(matchRow, 11).setValue('Y'); if (data.notes) sheet.getRange(matchRow, 14).setValue(data.notes); return ok('Trade Register updated — ' + data.trade); } else { var nextNum = rows.length; sheet.appendRow([ nextNum, data.trade || '', data.company || '', data.contact || '', data.phone || '', data.email || '', data.status || 'Quote Received', data.date || '', 'Yes', data.amount_inc_gst || '', 'Y', 'No', 'No', data.notes || '' ]); return ok('New trade added — ' + data.trade); } } if (action === 'update_cell') { var sheet2 = ss.getSheetByName(data.sheet); if (!sheet2) throw new Error('Sheet not found: ' + data.sheet); sheet2.getRange(data.row, data.col).setValue(data.value); return ok('Cell updated'); } if (action === 'append_row') { var sheet3 = ss.getSheetByName(data.sheet); if (!sheet3) throw new Error('Sheet not found: ' + data.sheet); sheet3.appendRow(data.values); return ok('Row appended'); } if (action === 'read_sheet') { var sheet4 = ss.getSheetByName(data.sheet); if (!sheet4) throw new Error('Sheet not found: ' + data.sheet); var values = sheet4.getDataRange().getValues(); return ContentService .createTextOutput(JSON.stringify({ status: 'success', data: values })) .setMimeType(ContentService.MimeType.JSON); } if (action === 'find_and_update') { var sheet5 = ss.getSheetByName(data.sheet); if (!sheet5) throw new Error('Sheet not found: ' + data.sheet); var vals = sheet5.getDataRange().getValues(); for (var i = 0; i < vals.length; i++) { if (String(vals[i][data.search_col - 1]).toLowerCase().indexOf(data.search_value.toLowerCase()) > -1) { sheet5.getRange(i + 1, data.update_col).setValue(data.update_value); return ok('Row ' + (i + 1) + ' updated'); } } throw new Error('Value not found: ' + data.search_value); } if (action === 'list_sheets') { var names = ss.getSheets().map(function(s) { return s.getName(); }); return ContentService .createTextOutput(JSON.stringify({ status: 'success', sheets: names })) .setMimeType(ContentService.MimeType.JSON); } throw new Error('Unknown action: ' + action); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', message: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var action = e.parameter.action || 'list_sheets'; if (action === 'list_sheets') { var names = ss.getSheets().map(function(s) { return s.getName(); }); return ContentService .createTextOutput(JSON.stringify({ status: 'success', sheets: names })) .setMimeType(ContentService.MimeType.JSON); } if (action === 'read_sheet') { var sheet = ss.getSheetByName(e.parameter.sheet); if (!sheet) throw new Error('Sheet not found'); var values = sheet.getDataRange().getValues(); return ContentService .createTextOutput(JSON.stringify({ status: 'success', data: values })) .setMimeType(ContentService.MimeType.JSON); } } catch(err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', message: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function ok(msg) { return ContentService .createTextOutput(JSON.stringify({ status: 'success', message: msg })) .setMimeType(ContentService.MimeType.JSON); }
